def hosts groups=nil
  
  groups = case groups
           when Array     then groups
           when String    then Array[groups]
           when NilClass  then @hostgroups
           end

  @hosts = []

  groups.each do |gr|
    @response = RestClient.get endpoint gr
    @hosts << JSON.parse(@response)['members'].split(',')
  end

  @hosts.flatten.uniq
end
